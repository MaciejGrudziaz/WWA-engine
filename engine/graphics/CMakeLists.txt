PROJECT(wwa-engine-graphics)

        init.cpp
)

ADD_LIBRARY(${PROJECT_NAME} SHARED ${SOURCE_FILES})

TARGET_LINK_LIBRARIES(${PROJECT_NAME} PUBLIC vulkan glfw)
